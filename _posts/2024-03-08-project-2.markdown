<script>
// Object to store user's choices for each question
let userChoices = {};

// Function to show the selected question
function showQuestion(q) {
  // Hide all questions
  const questions = document.querySelectorAll('.quiz-question');
  questions.forEach(function(qEl) {
    qEl.style.display = 'none';
  });

  // Show the selected question
  const selectedQuestion = document.getElementById('question-' + q);
  if (selectedQuestion) {
    selectedQuestion.style.display = 'block';
    
    // Refresh iframe content if needed
    const iframe = selectedQuestion.querySelector('iframe');
    if (iframe) {
      // Force reload the iframe
      const currentSrc = iframe.src;
      iframe.src = '';
      setTimeout(function() {
        iframe.src = currentSrc;
      }, 50);
    }
  }

  // Update button active state
  const buttons = document.querySelectorAll('.quiz-menu-button');
  buttons.forEach(function(btn, index) {
    btn.classList.remove('active');
    if (index === q - 1) btn.classList.add('active');
  });
}

// Function to handle user's choice selection
function selectChoice(choice, questionNumber, event) {
  userChoices[questionNumber] = choice;
  
  // Reset button styles
  const parentDiv = event.target.parentNode;
  const buttons = parentDiv.querySelectorAll('button');
  buttons.forEach(function(btn) {
    btn.style.opacity = '0.5';
  });
  
  // Highlight selected button
  event.target.style.opacity = '1';
}

// Function to confirm choice and show answer
function confirmChoice(questionNumber) {
  if (userChoices[questionNumber]) {
    // Show the answer
    const answerElement = document.getElementById('answer-' + questionNumber);
    if (answerElement) {
      answerElement.style.display = 'block';
    }
  } else {
    alert("Please select an answer first!");
  }
}

// Show the first question on page load
document.addEventListener("DOMContentLoaded", function() {
  showQuestion(1);
  
  // Set first button as active
  const firstButton = document.querySelector('.quiz-menu-button');
  if (firstButton) {
    firstButton.classList.add('active');
  }
});
</script>
